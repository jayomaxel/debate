## 1. 实时辩论流程编排

来源：

- `api/services/flow_controller.py`

这部分代码体现了平台如何把一场辩论拆成可执行的结构化流程，包括立论、盘问、自由辩论、总结陈词等阶段。

```python
class DebateFlowController:
    def initialize_flow(self, room_id: str, segments=None) -> None:
        self.segments[room_id] = segments or self._build_default_segments()
        self.segment_index[room_id] = 0

        room_state = room_manager.get_room_state(room_id)
        if room_state:
            room_state.flow_segments = self._serialize_segments(
                self.segments[room_id]
            )

    async def start_flow(self, room_id: str, segments=None) -> None:
        self.initialize_flow(room_id, segments)
        await self._apply_current_segment(room_id)
        await self.start_timer(room_id)

    def _build_default_segments(self) -> list[dict]:
        return [
            {
                "id": "opening_positive_1",
                "phase": DebatePhase.OPENING,
                "duration": 180,
                "mode": "fixed",
                "speaker_roles": ["debater_1"],
            },
            {
                "id": "questioning_1_ai2_ask",
                "phase": DebatePhase.QUESTIONING,
                "duration": 90,
                "mode": "fixed",
                "speaker_roles": ["ai_2"],
            },
            {
                "id": "free_debate",
                "phase": DebatePhase.FREE_DEBATE,
                "duration": 480,
                "mode": "free",
                "speaker_roles": [
                    "debater_1", "debater_2", "debater_3", "debater_4",
                    "ai_1", "ai_2", "ai_3", "ai_4",
                ],
            },
            {
                "id": "closing_positive_4",
                "phase": DebatePhase.CLOSING,
                "duration": 240,
                "mode": "fixed",
                "speaker_roles": ["debater_4"],
            },
        ]
```

可放在材料里的说明：

- 该模块将辩论过程抽象为 `segment` 序列，每个环节包含阶段、时长、发言模式和发言角色。
- 这种设计让不同赛制可以通过配置扩展，而不是把流程写死在前端页面里。
- 系统启动辩论时，会先装载流程，再切换到当前环节并启动计时器。

来源：

- `api/services/room_manager.py`

这部分代码体现了平台如何维护“一个实时辩论房间”的完整运行状态，是前后端实时协同的关键。

```python
class DebatePhase(str, Enum):
    WAITING = "waiting"

    QUESTIONING = "questioning"

class RoomState:
    def __init__(

        current_phase: DebatePhase = DebatePhase.WAITING,

        segment_id: str | None = None,
        speaker_mode: str | None = None,
        speaker_options: list[str] | None = None,

        ai_turn_status: str = "idle",
        playback_gate_status: str = "idle",
        timer_status: str = "idle",
        room_capacity: int = 4,
        required_roles: list[str] | None = None,
        participants: list[dict] | None = None,
        ai_debaters: list[dict] | None = None,
    ):
        self.room_id = room_id
        self.debate_id = debate_id
        self.current_phase = current_phase
        self.time_remaining = time_remaining
        self.current_speaker = current_speaker
        self.segment_index = segment_index
        self.segment_id = segment_id
        self.speaker_mode = speaker_mode
        self.speaker_options = speaker_options or []
        self.mic_owner_user_id = mic_owner_user_id
        self.free_debate_next_side = free_debate_next_side or "human"
        self.ai_turn_status = ai_turn_status
        self.playback_gate_status = playback_gate_status
        self.timer_status = timer_status
        self.room_capacity = room_capacity
        self.required_roles = required_roles or ["debater_1", "debater_2", "debater_3", "debater_4"]
        self.participants = participants or []
        self.ai_debaters = ai_debaters or []

    def to_dict(self) -> dict:
        return {
            "current_phase": self.current_phase,
            "time_remaining": self.time_remaining,
            "current_speaker": self.current_speaker,
            "segment_index": self.segment_index,
            "segment_id": self.segment_id,
            "speaker_mode": self.speaker_mode,
            "speaker_options": self.speaker_options,
            "mic_owner_user_id": self.mic_owner_user_id,
            "free_debate_next_side": self.free_debate_next_side,
            "ai_turn_status": self.ai_turn_status,
            "playback_gate_status": self.playback_gate_status,
            "timer_status": self.timer_status,
            "participants": self.participants,
            "ai_debaters": self.ai_debaters,
        }
```

可放在材料里的说明：

- 该状态模型统一管理发言人、计时器、麦克风占用、AI 回合、音频播放门控等信息。
- 所有前端界面都围绕同一个房间状态工作，因此学生端、教师端、评审端看到的是同一份实时数据。
- `to_dict()` 的作用是把运行中的 Python 对象转成可通过 WebSocket 广播的结构化 JSON。

## 3. 基于 pgvector 的知识库检索增强问答

来源：

- `api/services/rag_service.py`

这部分代码体现了平台不是简单调用大模型回答问题，而是先检索课程知识库，再生成有依据的答案。

### 3.1 向量检索核心逻辑

```python
async def search_similar_chunks(
    self,
    query_embedding: list[float],
    top_k: int | None = None,
    similarity_threshold: float | None = None
) -> list[dict]:
    top_k = top_k or self.top_k
    similarity_threshold = similarity_threshold or self.similarity_threshold

    vector_str = "[" + ",".join(str(x) for x in query_embedding) + "]"

    query = text("""
        SELECT
            c.id as chunk_id,
            c.document_id,
            c.content,
            c.chunk_index,
            c.token_count,
            d.filename as document_name,
            1 - (c.embedding <=> CAST(:query_vector AS vector)) as similarity_score
        FROM kb_document_chunks c
        JOIN kb_documents d ON c.document_id = d.id
        WHERE d.upload_status = 'completed'
          AND c.embedding IS NOT NULL
          AND 1 - (c.embedding <=> CAST(:query_vector AS vector)) >= :threshold
        ORDER BY c.embedding <=> CAST(:query_vector AS vector)
        LIMIT :limit
    """)

    result = self.db.execute(
        query,
        {
            "query_vector": vector_str,
            "threshold": similarity_threshold,
            "limit": top_k,
        },
    )

    return [
        {
            "chunk_id": str(row.chunk_id),
            "document_id": str(row.document_id),
            "document_name": row.document_name,
            "content": row.content,
            "chunk_index": row.chunk_index,
            "token_count": row.token_count,
            "similarity_score": float(row.similarity_score),
        }
        for row in result
    ]
```

可放在材料里的说明：

- 项目使用 `pgvector` 在 PostgreSQL 中直接完成向量相似度搜索。
- 检索结果不是整篇文档，而是切分后的知识块 `chunk`，能提高回答的准确性和引用粒度。
- `similarity_threshold` 用于过滤低相关内容，避免模型“拿错资料”。

### 3.2 问答主流程

```python
async def ask_question(self, question: str, user_id: str, session_id: str) -> dict:
    query_embedding = await self.generate_query_embedding(question)

    similar_chunks = await self.search_similar_chunks(
        query_embedding=query_embedding,
        top_k=self.top_k,
        similarity_threshold=self.similarity_threshold,
    )

    answer = await self.generate_answer(
        question=question,
        context_chunks=similar_chunks,
    )

    sources = self.format_source_citations(similar_chunks)
    used_kb = len(sources) > 0

    conversation = KBConversation(
        id=uuid.uuid4(),
        user_id=uuid.UUID(user_id),
        session_id=session_id,
        question=question,
        answer=answer,
        sources=sources,
        created_at=datetime.utcnow(),
    )
    self.db.add(conversation)
    self.db.commit()

    return {
        "answer": answer,
        "sources": sources,
        "used_kb": used_kb,
    }
```

可放在材料里的说明：

- 整体链路为“问题嵌入化 -> 相似片段检索 -> 大模型生成 -> 引用返回 -> 对话留痕”。
- 这使知识库问答不仅能回答学生问题，还能保留会话记录，便于后续教学分析。
- 返回结果里带有 `sources`，可以在前端展示引用来源，增强可解释性。

### 4.1 学生报告数据聚合

class ReportGenerator:
    @staticmethod
    def generate_student_report(db: Session, debate_id: str, student_id: str):
        debate = db.execute(
            select(Debate).where(Debate.id == debate_id)
        ).scalar_one_or_none()

        speeches = (
            db.execute(
                select(Speech)
                .where(Speech.debate_id == debate_id)
                .where(Speech.is_valid_for_scoring.is_(True))
                .order_by(Speech.timestamp)
            )
            .scalars()
            .all()
        )

        score_rows = (
            db.execute(
                select(Score, Speech)

                .where(Speech.debate_id == debate_id)

            .all()

### 4.2 报告导出 Markdown 骨架

```python
@staticmethod
def _build_fallback_markdown(report: Report) -> str:
    participants_md = "\n".join(
        [
            f"- {p.get('name')}（{p.get('role')}）得分："
            f"{(p.get('final_score') or {}).get('overall_score', 0)}"
            for p in (report.participants or [])
        ]
    )

    speeches_md_lines = []
    for i, s in enumerate(report.speeches or []):
        score = (s.get("score") or {}).get("overall_score") if s.get("score") else None
        speaker_name = s.get("speaker_name") or s.get("speaker_role") or "未知"
        phase = s.get("phase") or ""
        speeches_md_lines.append(
            f"{i+1}. **[{phase}] {speaker_name}**（得分：{score}）\n\n"
            f"{s.get('content') or ''}\n"
        )

    speeches_md = "\n".join(speeches_md_lines)

    return (
        f"# 辩论报告\n\n"
        f"## 基本信息\n\n"
        f"- 辩题：{report.topic}\n"
        f"- 开始时间：{report.start_time}\n"
        f"- 结束时间：{report.end_time}\n"
        f"- 持续时间：{report.duration}分钟\n"
        f"- 获胜方：{report.winner}\n\n"
        f"## 参与者\n\n"
        f"{participants_md}\n\n"
        f"## 发言记录\n\n"
        f"{speeches_md}\n"
    )
```

可放在材料里的说明：

- 系统可自动把结构化比赛数据转成标准 Markdown 报告，再进一步导出 PDF。
- 即使大模型报告生成链路不可用，也有后端模板化兜底方案，保证报告稳定产出。
- 这种设计兼顾了“智能生成效果”和“工程可用性”。
